// Cloudflare Worker — VN Stock Intelligence // Thêm endpoint /news/{ticker} để fetch Cafef RSS/search (tránh CORS) // Deploy: wrangler deploy hoặc paste vào Cloudflare Dashboard const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET,OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', } export default { async fetch(request, env, ctx) { const url = new URL(request.url) const path = url.pathname // e.g. /VCB or /news/VCB or /events/VCB if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: CORS }) } // ── /news/{ticker} ───────────────────────────────────────────── if (path.startsWith('/news/')) { const ticker = path.split('/')[2]?.toUpperCase() if (!ticker) return jsonResp([], 400) return await fetchNews(ticker) } // ── /events/{ticker} ────────────────────────────────────────── if (path.startsWith('/events/')) { const ticker = path.split('/')[2]?.toUpperCase() if (!ticker) return jsonResp([], 400) return await fetchEvents(ticker) } // ── /info/{ticker} ──────────────────────────────────────────── if (path.startsWith('/info/')) { const ticker = path.split('/')[2]?.toUpperCase() if (!ticker) return jsonResp({ error: 'no ticker' }, 400) return await fetchInfo(ticker) } // ── /fundamental/{ticker} → VNDirect finfo proxy ───────────── if (path.startsWith('/fundamental/')) { const ticker = path.split('/')[2]?.toUpperCase() if (!ticker) return jsonResp({ error: 'no ticker' }, 400) return await fetchFundamental(ticker) } // ── /{ticker} → Yahoo Finance (giá real-time) ───────────────── // Hỗ trợ cả VN stocks (.VN) và commodity symbols (BZ=F, GC=F, USDVND=X) const rawSym = path.slice(1).split('?')[0] const ticker = rawSym.toUpperCase() if (!ticker || ticker.includes('/')) return jsonResp({ error: 'invalid' }, 400) // Commodity symbols: giữ nguyên, VN stocks: thêm .VN const isCommodity = ticker.includes('=') || ticker.includes('-') return await fetchYahoo(ticker, isCommodity) } } // ── Fetch Yahoo Finance ──────────────────────────────────────────────── async function fetchYahoo(ticker, isCommodity = false) { // Commodity symbols (BZ=F, GC=F, USDVND=X) dùng nguyên symbol // VN stocks thêm .VN suffix const symbol = isCommodity ? ticker : `${ticker}.VN` const yUrl = `https://query1.finance.yahoo.com/v8/finance/chart/${encodeURIComponent(symbol)}?interval=1d&range=90d&includePrePost=false` try { const r = await fetch(yUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Accept': 'application/json', 'Accept-Language': 'en-US,en;q=0.9', }, cf: { cacheTtl: isCommodity ? 900 : 300, cacheEverything: true } }) if (!r.ok) return jsonResp({ error: `Yahoo ${r.status}` }, 502) const d = await r.json() return jsonResp({ source: 'yahoo', ...d }) } catch(e) { return jsonResp({ error: e.message }, 500) } } // ── Fetch Fundamental (VNDirect proxy) ─────────────────────────────── async function fetchFundamental(ticker) { // VNDirect finfo API — chỉ accessible từ server (CORS block browser) const url = `https://finfo-api.vndirect.com.vn/v4/ratio/latest?q=code:${ticker}&fields=priceToEarning,priceToBook,returnOnEquity,earningPerShare,dividendYield,marketCap` try { const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Accept': 'application/json', 'Referer': 'https://www.vndirect.com.vn/', }, cf: { cacheTtl: 3600, cacheEverything: true } }) if (!r.ok) return jsonResp({ error: `VNDirect ${r.status}` }, 502) const d = await r.json() const item = d?.data?.[0] if (!item) return jsonResp({ error: 'no data' }, 404) return jsonResp({ pe: item.priceToEarning || null, pb: item.priceToBook || null, roe: item.returnOnEquity || null, eps: item.earningPerShare || null, dividendYield: item.dividendYield || null, marketCap: item.marketCap || null, }) } catch(e) { return jsonResp({ error: e.message }, 500) } } // ── Fetch Company Info ──────────────────────────────────────────────── async function fetchInfo(ticker) { // VNDirect company info API const apiUrl = `https://finfo-api.vndirect.com.vn/v4/company_profiles?q=code:${ticker}&fields=code,companyName,shortName,floor,industryName,icbName` try { const r = await fetch(apiUrl, { headers: { 'User-Agent': 'Mozilla/5.0', 'Accept': 'application/json' }, cf: { cacheTtl: 3600, cacheEverything: true } }) if (!r.ok) return jsonResp({ error: 'info not found' }, 404) const d = await r.json() const item = d?.data?.[0] if (!item) return jsonResp({ error: 'not found' }, 404) return jsonResp(item) } catch(e) { return jsonResp({ error: e.message }, 500) } } // ── Fetch Events ──────────────────────────────────────────────────── async function fetchEvents(ticker) { // VNDirect events API (cổ tức, ĐHCĐ) const apiUrl = `https://finfo-api.vndirect.com.vn/v4/dividends?q=code:${ticker}&sort=exDividendDate:desc&size=10` try { const r = await fetch(apiUrl, { headers: { 'User-Agent': 'Mozilla/5.0', 'Accept': 'application/json' }, cf: { cacheTtl: 3600, cacheEverything: true } }) if (!r.ok) return jsonResp([], 200) const d = await r.json() const events = (d?.data || []).map(e => ({ type: 'Cổ tức', date: e.exDividendDate || e.recordDate, detail: e.cashDividendPercentage ? `${e.cashDividendPercentage}%` : (e.dividendType || ''), })) return jsonResp(events) } catch(e) { return jsonResp([], 200) } } // ── Fetch News ──────────────────────────────────────────────────────── async function fetchNews(ticker) { // Thử 3 nguồn song song: CafeF RSS, VNDirect news, Google News RSS const results = await Promise.allSettled([ fetchCafefNews(ticker), fetchVnDirectNews(ticker), fetchGoogleNews(ticker), ]) const allNews = [] for (const r of results) { if (r.status === 'fulfilled' && Array.isArray(r.value)) { allNews.push(...r.value) } } // Dedupe theo title const seen = new Set() const unique = allNews.filter(n => { const key = n.title?.slice(0, 40) if (seen.has(key)) return false seen.add(key) return true }) // Sort by date desc, lấy 10 tin const sorted = unique .sort((a, b) => new Date(b.date || 0) - new Date(a.date || 0)) .slice(0, 10) return jsonResp(sorted) } async function fetchCafefNews(ticker) { // CafeF RSS feed cho mã cụ thể const url = `https://cafef.vn/du-lieu/Ajax/PageNew/LoadDataNewsByStockCode.ashx?stockCode=${ticker}&pageIndex=1&pageSize=10` try { const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Referer': 'https://cafef.vn/', 'Accept': 'application/json, text/javascript, */*' }, cf: { cacheTtl: 1800, cacheEverything: true } }) if (!r.ok) return [] const text = await r.text() let data try { data = JSON.parse(text) } catch { return [] } const items = data?.Data?.ListNews || data?.data || [] return items.slice(0, 8).map(item => ({ title: item.Title || item.title || item.Name || '', url: item.Url ? `https://cafef.vn${item.Url}` : (item.url || null), date: item.PublishDate || item.publishDate || item.CreateDate || '', source: 'CafeF', })).filter(n => n.title) } catch(e) { return [] } } async function fetchVnDirectNews(ticker) { const url = `https://finfo-api.vndirect.com.vn/v4/news?q=symbol:${ticker}&sort=date:desc&size=8` try { const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0', 'Accept': 'application/json' }, cf: { cacheTtl: 1800, cacheEverything: true } }) if (!r.ok) return [] const d = await r.json() return (d?.data || []).map(item => ({ title: item.title || item.name || '', url: item.url || null, date: item.date || item.publishedDate || '', source: 'VNDirect', })).filter(n => n.title) } catch(e) { return [] } } async function fetchGoogleNews(ticker) { // Google News RSS — tìm theo tên mã tiếng Việt const query = encodeURIComponent(`cổ phiếu ${ticker}`) const url = `https://news.google.com/rss/search?q=${query}&hl=vi&gl=VN&ceid=VN:vi` try { const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0' }, cf: { cacheTtl: 1800, cacheEverything: true } }) if (!r.ok) return [] const xml = await r.text() const items = [] const itemRegex = /([\s\S]*?)<\/item>/g let match while ((match = itemRegex.exec(xml)) !== null && items.length < 6) { const item = match[1] const title = (/<!\[CDATA\[(.*?)\]\]><\/title>/.exec(item) || /<title>(.*?)<\/title>/.exec(item))?.[1] || '' // Google News: URL thật nằm trong <guid> hoặc sau source name trong <link> const guid = (/<guid[^>]*>(.*?)<\/guid>/.exec(item))?.[1] || '' const linkRaw = (/<link>(.*?)<\/link>/.exec(item))?.[1] || '' // Ưu tiên guid nếu là URL thật (bắt đầu http và không phải google.com/articles) let finalUrl = null if (guid.startsWith('http') && !guid.includes('news.google.com')) { finalUrl = guid } else if (linkRaw.startsWith('http') && !linkRaw.includes('news.google.com')) { finalUrl = linkRaw } else { // Fallback: dùng Google redirect link (user vẫn đến được tin) finalUrl = linkRaw || guid || null } const pubDate = (/<pubDate>(.*?)<\/pubDate>/.exec(item))?.[1] || '' if (title) items.push({ title: title.trim(), url: finalUrl, date: pubDate, source: 'Google News' }) } return items } catch(e) { return [] } } // ── Helper ──────────────────────────────────────────────────────────── function jsonResp(data, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...CORS, 'Content-Type': 'application/json' } }) }
